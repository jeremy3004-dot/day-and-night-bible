    2. Create `src/services/queryClient.ts`:
       ```typescript
       import { QueryClient } from '@tanstack/react-query';
       import { focusManager, onlineManager } from '@tanstack/react-query';
       import { AppState, type AppStateStatus } from 'react-native';
       import NetInfo from '@react-native-community/netinfo';

       // Re-fetch stale queries when app returns to foreground
       function onAppStateChange(status: AppStateStatus) {
         focusManager.setFocused(status === 'active');
       }

       // Track online/offline status via NetInfo (already installed)
       onlineManager.setEventListener((setOnline) => {
         return NetInfo.addEventListener((state) => {
           setOnline(!!state.isConnected);
         });
       });

       // Listen for app state changes
       AppState.addEventListener('change', onAppStateChange);

       export const queryClient = new QueryClient({
         defaultOptions: {
           queries: {
             retry: 2,
             staleTime: 5 * 60 * 1000, // 5 minutes
             gcTime: 10 * 60 * 1000, // 10 minutes (garbage collect)
           },
         },
       });
       ```

    Notes:
    - Use `focusManager.setFocused()` (not `focusManager.setEventListener`) for simpler AppState integration per TanStack RN docs.
    - `gcTime` replaces the old `cacheTime` (renamed in TanStack Query v5).
    - `@react-native-community/netinfo` is already in package.json — no additional native module install.
    - Do NOT import React in this file — it is a pure setup module.
  </action>
